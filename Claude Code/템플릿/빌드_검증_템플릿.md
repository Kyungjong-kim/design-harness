# 하네스 빌드 검증 템플릿

`/project-init` 완료 후 실제 생성 파일과 기대 산출물을 비교한다.
누락(✗) 항목 발견 시 아래 "보완 경로" 표를 참고해 해당 Step으로 돌아가 즉시 보완 후 재실행.

---

## 검증 스크립트

```bash
# ====================================================
# 사용 전 아래 변수를 실제 값으로 교체
SERVICE="<서비스명>"         # Q1 답변 (예: my-portal)
AGENT_DIR="<에이전트경로>"   # Q4 답변: ~/.claude/agents/ 또는 .claude/agents/
PROJECT_TYPE="FE"            # FE / BE / 풀스택 (Step 1 분석 결과)
# DOCS_BASE: Step 3 $BASE 와 동일한 값으로 설정
# 단일 앱 / 모노레포 루트 통합형: DOCS_BASE="docs/$SERVICE"
# 모노레포 패키지별 (패키지 루트에서 실행): DOCS_BASE="docs"
DOCS_BASE="docs/$SERVICE"
# ====================================================

PASS=0; FAIL=0

check_file() {
  local label="$1"; local path="$2"
  if [ -f "$path" ]; then
    if [ -s "$path" ]; then
      echo "  ✓ $label"
    else
      echo "  ⚠ $label — 파일 있으나 비어 있음"
    fi
    PASS=$((PASS+1))
  else
    echo "  ✗ $label  ←  $path"
    FAIL=$((FAIL+1))
  fi
}

warn_file() {
  local label="$1"; local path="$2"
  if [ -f "$path" ] && [ -s "$path" ]; then
    echo "  ✓ $label"
  elif [ -f "$path" ]; then
    echo "  ⚠ $label — 비어 있음"
  else
    echo "  ⚠ $label — 없음 (FE·풀스택이면 Step 5-D/9-C로 돌아가 보완)"
  fi
}

echo "=============================="
echo "  하네스 빌드 검증"
echo "=============================="

echo ""
echo "[필수 파일]"
check_file "CLAUDE.md"             "CLAUDE.md"
check_file "HANDOFF_NOW.md"        "$DOCS_BASE/status/HANDOFF_NOW.md"
check_file "HANDOFF.md"            "$DOCS_BASE/plans/HANDOFF.md"
check_file "세션_노트.md"           "$DOCS_BASE/history/세션_노트.md"
check_file "agent/README.md"       "$DOCS_BASE/agent/README.md"
check_file "agent/architecture.md" "$DOCS_BASE/agent/architecture.md"
check_file "agent/conventions.md"  "$DOCS_BASE/agent/conventions.md"
check_file "branch-commit.md"      "$DOCS_BASE/git-workflow/branch-commit.md"

echo ""
echo "[에이전트]"
check_file "${SERVICE}-dev.md"        "$AGENT_DIR/${SERVICE}-dev.md"
check_file "${SERVICE}-doc-writer.md" "$AGENT_DIR/${SERVICE}-doc-writer.md"

echo ""
echo "[스킬]"
for skill in session-close project-fix project-pr project-issue document-review; do
  if [ -d "$HOME/.claude/skills/$skill" ] || [ -d ".claude/skills/$skill" ]; then
    echo "  ✓ /$skill"
    PASS=$((PASS+1))
  else
    echo "  ✗ /$skill — 미설치"
    FAIL=$((FAIL+1))
  fi
done

if [ "$PROJECT_TYPE" != "BE" ]; then
  echo ""
  echo "[조건부 — FE·풀스택]"
  warn_file "agent/design-system.md" "$DOCS_BASE/agent/design-system.md"
  FE_COUNT=$(find "$DOCS_BASE/pages" -name "FE_*.md" 2>/dev/null | wc -l | tr -d ' ')
  if [ "$FE_COUNT" -gt 0 ]; then
    echo "  ✓ FE_*.md — ${FE_COUNT}개"
  else
    echo "  ⚠ FE_*.md — $DOCS_BASE/pages/ 하위 없음 (Step 9-C로 돌아가 보완)"
  fi
fi

if [ "$PROJECT_TYPE" != "FE" ]; then
  echo ""
  echo "[조건부 — BE·풀스택]"

  # BE 기능 문서 확인
  BE_COUNT=$(find "$DOCS_BASE/pages" -name "BE_*.md" 2>/dev/null | wc -l | tr -d ' ')
  if [ "$BE_COUNT" -gt 0 ]; then
    echo "  ✓ BE_*.md — ${BE_COUNT}개"
  else
    echo "  ⚠ BE_*.md — $DOCS_BASE/pages/ 하위 없음 (Step 9-C로 돌아가 보완)"
  fi

  # DB 마이그레이션 폴더 확인
  MIGRATION_DIR=$(find . -maxdepth 5 -type d \( -name "migrations" -o -name "alembic" -o -name "migrate" \) \
    -not -path "*/node_modules/*" 2>/dev/null | head -1)
  if [ -n "$MIGRATION_DIR" ]; then
    echo "  ✓ 마이그레이션 디렉토리: $MIGRATION_DIR"
  else
    echo "  ⚠ 마이그레이션 디렉토리 없음 — DB 없는 프로젝트면 무시"
  fi

  # 테스트 폴더 확인
  TEST_DIR=$(find . -maxdepth 4 -type d \( -name "tests" -o -name "test" -o -name "__tests__" \) \
    -not -path "*/node_modules/*" 2>/dev/null | head -1)
  if [ -n "$TEST_DIR" ]; then
    echo "  ✓ 테스트 디렉토리: $TEST_DIR"
  else
    echo "  ⚠ 테스트 디렉토리 없음"
  fi

  # BE 빌드·서버 기동 확인 안내 (프레임워크별 명령 분기)
  LANG_HINT=""
  if [ -f "pyproject.toml" ] || [ -f "requirements.txt" ]; then LANG_HINT="python"
  elif [ -f "go.mod" ]; then LANG_HINT="go"
  elif [ -f "package.json" ]; then LANG_HINT="node"
  fi

  echo ""
  echo "  [BE 수동 확인 항목]"
  case "$LANG_HINT" in
    python)
      echo "  □ 서버 기동: uvicorn app.main:app --reload  또는  python -m flask run"
      echo "  □ 헬스체크: curl localhost:8000/health"
      echo "  □ 단위 테스트: pytest tests/ -v"
      echo "  □ DB 마이그레이션: alembic current  /  alembic upgrade head"
      ;;
    go)
      echo "  □ 서버 기동: go run ./cmd/server  또는  go run ."
      echo "  □ 헬스체크: curl localhost:8080/api/health"
      echo "  □ 단위 테스트: go test ./..."
      echo "  □ DB 마이그레이션: migrate -path ./migrations -database \"\$DB_URL\" up"
      ;;
    node)
      echo "  □ 서버 기동: npm run dev  또는  npm start"
      echo "  □ 헬스체크: curl localhost:3000/health"
      echo "  □ 단위 테스트: npm test"
      echo "  □ DB 마이그레이션: npx prisma migrate dev  /  직접 확인"
      ;;
    *)
      echo "  □ 서버 기동: <서버 실행 명령>"
      echo "  □ 헬스체크: curl localhost:<포트>/health"
      echo "  □ 단위 테스트: pytest / go test ./... / npm test"
      echo "  □ DB 마이그레이션: alembic current / migrate up / 직접 확인"
      ;;
  esac
fi

echo ""
echo "=============================="
if [ "$FAIL" -eq 0 ]; then
  echo "  결과: 필수 항목 전량 통과 ✓  (${PASS}개)"
else
  echo "  결과: ✓ ${PASS}개 통과  /  ✗ ${FAIL}개 누락"
  echo "  → 아래 '보완 경로' 표 참고 후 해당 Step으로 돌아가 보완, 재실행"
fi
echo "=============================="
```

| [서비스명]-dev.md · doc-writer.md | Step 6 | Phase 6 |

| HANDOFF_NOW.md · HANDOFF.md · 세션_노트.md | Step 9 | Phase 9-A |
| branch-commit.md | Step 9 | Phase 9-B |
| agent/design-system.md | Step 5 | Phase 5-D |
| FE_*.md | Step 9 | Phase 9-C |
| BE_*.md | Step 9 | Phase 9-C |
